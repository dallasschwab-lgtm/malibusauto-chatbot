<script>

  var BRAND_COLOR = "#c8102e";
  var BOT_NAME = "Malibu's Auto";

    "#mac-chat-messages { flex:1; overflow-y:auto; padding:16px; display:flex; flex-direction:column; gap:10px; }",
    ".mac-msg { max-width:80%; padding:10px 14px; border-radius:14px; line-height:1.5; }",
    ".mac-msg.bot { background:#f0f0f0; color:#222; align-self:flex-start; border-bottom-left-radius:4px; }",
    ".mac-msg.user { background:" + BRAND_COLOR + "; color:white; align-self:flex-end; border-bottom-right-radius:4px; }",
    ".mac-typing { display:flex; gap:4px; align-items:center; padding:10px 14px; }",
    ".mac-typing span { width:7px; height:7px; background:#aaa; border-radius:50%; animation:mac-bounce .8s infinite; }",

  win.id = "mac-chat-window";
  win.innerHTML = '<div id="mac-chat-header"><div class="avatar">🔧</div><div><div>' + BOT_NAME + '</div><div style="font-size:11px;font-weight:normal;opacity:.85">Accessories &amp; Installation</div></div><div id="mac-chat-close">✕</div></div><div id="mac-chat-messages"></div><div id="mac-chat-input-row"><textarea id="mac-chat-input" placeholder="Ask about your vehicle, services, hours..."></textarea><button id="mac-chat-send"><svg viewBox="0 0 24 24"><path d="M2.01 21L23 12 2.01 3 2 10l15 2-15 2z"/></svg></button></div>';

  document.body.appendChild(bubble);
  document.body.appendChild(win);

  var msgs = document.getElementById("mac-chat-messages");
  var input = document.getElementById("mac-chat-input");
  var isOpen = false;

  function toggleChat() {
    isOpen = !isOpen;
    win.classList.toggle("open", isOpen);
    if (isOpen && msgs.children.length === 0) addBotMsg("Hi! 👋 Welcome to Malibu's Auto Center. How can I help you today?");
  }

  function addBotMsg(text) {
    var el = document.createElement("div");
    el.className = "mac-msg bot";
    el.textContent = text;
    msgs.appendChild(el);
    msgs.scrollTop = msgs.scrollHeight;
  }

  function addUserMsg(text) {
    var el = document.createElement("div");
    el.className = "mac-msg user";
    el.textContent = text;
    msgs.appendChild(el);
    msgs.scrollTop = msgs.scrollHeight;
  }

  function showTyping() {
    var el = document.createElement("div");
    el.className = "mac-msg bot mac-typing";
    el.id = "mac-typing";
    el.innerHTML = "<span></span><span></span><span></span>";
    msgs.appendChild(el);
    msgs.scrollTop = msgs.scrollHeight;
    return el;
  }

  function sendMessage() {
    var text = input.value.trim();
    if (!text) return;
    input.value = "";
    addUserMsg(text);
    var typing = showTyping();
    fetch(CHATBOT_URL, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ message: text, sessionId: sessionId })
    })
    .then(function(res) { return res.json(); })
    .then(function(data) {
      typing.remove();
      addBotMsg(data.reply || "Sorry, I had trouble responding. Please call us directly!");
    })
    .catch(function() {
      typing.remove();
      addBotMsg("Having trouble connecting. Please call us at 405-799-6700!");
    });
  }

  bubble.addEventListener("click", toggleChat);
  document.getElementById("mac-chat-close").addEventListener("click", toggleChat);
  document.getElementById("mac-chat-send").addEventListener("click", sendMessage);
  input.addEventListener("keydown", function(e) {
    if (e.key === "Enter" && !e.shiftKey) { e.preventDefault(); sendMessage(); }
  });
})();
</script>
</body>
</html>
