2. In the GET handler, change it so that GET / serves the widget HTML, and any other GET path returns the health JSON:

Replace this:
  // ── GET — health check ──
  if (req.method === "GET") {
    res.writeHead(200, { "Content-Type": "application/json" });
    res.end(JSON.stringify({ name: "shopmonkey-mcp", version: "1.0.0", protocol: "mcp", status: "ok", chatbot: "enabled" }));
    return;
  }

With this:
  // ── GET — widget or health check ──
  if (req.method === "GET") {
    if (pathname === "/" || pathname === "") {
      res.writeHead(200, { "Content-Type": "text/html" });
      res.end(WIDGET_HTML);
    } else {
      res.writeHead(200, { "Content-Type": "application/json" });
      res.end(JSON.stringify({ name: "shopmonkey-mcp", version: "1.0.0", protocol: "mcp", status: "ok", chatbot: "enabled" }));
    }
    return;
  }

Note: the CHATBOT_URL in the widget is set to "/chat" (relative path) so it will automatically use the same server — no hardcoded URL needed.
